def sum_or_product(end_num, sum=true)
  if sum 
    start_num = 1
    num_array = []
    while start_num <= end_num
      num_array.push(start_num)
      start_num += 1
    end
    sum = 0
    num_array.each {|num| sum += num }
    puts sum
  else 
    start_num = 1
    num_array = []
    while start_num <= end_num
      num_array.push(start_num)
      start_num += 1
    end

  end
end
